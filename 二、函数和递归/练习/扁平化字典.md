## 扁平化字典


```python
source = {'a': {'b': 1,'c': 2}, 'd': {'e': 3,'f': {'g': 4}},'h': 5}
def flatmap(src,prefix='',target={}):
    for k,v in src.items():
        if isinstance(v,dict):
            flatmap(v,k,target)
        else:
            target[prefix+'.'+k] = v
    return target
print(flatmap(source))

结果如下
{'a.b': 1, 'a.c': 2, 'd.e': 3, 'f.g': 4, '.h': 5}
```

修改.的位置

```python
source = {'a': {'b': 1,'c': 2}, 'd': {'e': 3,'f': {'g': 4}},'h': 5}
def flatmap(src,prefix='',target=None):
    if target is None:
        target = {}
    for k,v in src.items():
        if isinstance(v,dict):
            flatmap(v,prefix+k+'.',target)
        else:
            target[prefix+k] = v
    return target
print(flatmap(source))
```

然后，这些无关参数最好封装到内部

```Python
source = {'a': {'b': 1,'c': 2}, 'd': {'e': 3,'f': {'g': 4}},'h': 5}
def flatmap(src):
    prefix = ''
    dct = {}
    def inner(src,prefix):
        for k,v in src.items():
            if isinstance(v, dict):
                inner(v,prefix+k+'.')
            else:
                dct[prefix+k]=v
    inner(src,prefix)
    return dct
print(flatmap(src=source))

{'a.b': 1, 'a.c': 2, 'd.e': 3, 'd.f.g': 4, 'h': 5}
```
